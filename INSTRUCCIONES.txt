     - Copiar wow_theme.py
     - Copiar usuarios.json
     - Copiar requirements.txt

     - Copiar README.md
     
  4. En PowerShell:
     git add .
     git commit -m "OPTIMIZACIÓN FINAL: Limpieza, asistencia completa, 3 módulos funcionales"
     git push origin main
     
  5. En VPS:
     cd /opt/Proyecto_Jerarquia
     git pull origin main
     systemctl restart proyecto_jerarquia
     
  6. Probar:
     http://2.25.171.171:8501
